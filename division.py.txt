import random

def division_game():
num1 = random.randint(1, 100)
num2 = random.randint(1, 10)
    
result = num1 / num2
guess = float(input(f"¿Cuál es el resultado de {num1} / {num2}? "))
    
if guess == result:
print("¡Correcto!")
else:
print(f"Incorrecto. El resultado correcto es {result}.")

division_game()
